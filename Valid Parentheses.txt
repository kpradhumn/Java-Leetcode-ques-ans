//20.Valid Parentheses


class Solution {
    public boolean isValid(String s) {
        Stack<Character> stack = new Stack<>();
        for(char ch : s.toCharArray()){
            if(ch == '(' || ch == '[' || ch == '{')
            stack.push(ch);
            else{
                if(stack.isEmpty()){
                    return false;
                }
                    char n = stack.pop();
                    if((n == '(' && ch !=')') ||
                       (n == '{' && ch !='}') ||
                       (n == '[' && ch !=']')  ){
                        return false;
                       }
            }
        }
        return stack.isEmpty();
        
    }
}
